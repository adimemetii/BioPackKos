        // Initialize translations
        function initTranslations() {
            // Update title
            document.title = translations[currentLang]['catalog-page-title'];
            
            // Update all translatable elements
            document.querySelectorAll('[data-translate]').forEach(element => {
                const key = element.getAttribute('data-translate');
                if (translations[currentLang] && translations[currentLang][key]) {
                    element.textContent = translations[currentLang][key];
                }
            });
        }

        // Language toggle
        function toggleLanguage() {
            // Language rotation order: EN -> DE -> FR -> IT -> SQ
            const languages = ['en', 'de', 'fr', 'it', 'sq'];
            
            // Get current index
            let currentIndex = languages.indexOf(currentLang);
            
            // Move to next language
            currentIndex = (currentIndex + 1) % languages.length;
            currentLang = languages[currentIndex];
            
            localStorage.setItem('language', currentLang);
            document.getElementById('lang-toggle').textContent = languageLabels[currentLang];
            document.documentElement.lang = currentLang;
           
            // Update title
            document.title = translations[currentLang]['catalog-page-title'];
           
            // Update all translatable elements
            document.querySelectorAll('[data-translate]').forEach(element => {
                const key = element.getAttribute('data-translate');
                if (translations[currentLang] && translations[currentLang][key]) {
                    element.textContent = translations[currentLang][key];
                }
            });
        }

        // Initialize on page load
        initTranslations();

        function openModal(imageSrc, number) {
            const modal = document.getElementById('catalogModal');
            const modalImage = document.getElementById('modalImage');
            const catalogText = currentLang === 'sq' ? 'Katalog' : 'Catalog';
            modalImage.src = imageSrc;
            modalImage.alt = `${catalogText} ${number}`;
            modal.classList.add('active');
            document.body.style.overflow = 'hidden';
        }

        function closeModal() {
            const modal = document.getElementById('catalogModal');
            modal.classList.remove('active');
            document.body.style.overflow = 'auto';
        }

        // Close modal on ESC key
        document.addEventListener('keydown', function(e) {
            if (e.key === 'Escape') {

            }
        });

        // Close modal when clicking outside

            if (e.target === this) {
